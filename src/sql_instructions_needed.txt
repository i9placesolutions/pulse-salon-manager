
-- Observe que precisamos criar várias funções RPC para substituir os acessos diretos às tabelas.
-- Isso soluciona os problemas de tipagem do TypeScript com tabelas que não estão no esquema de tipos oficial.

-- 1. Função para inserir eventos de webhook
CREATE OR REPLACE FUNCTION public.insert_webhook_event(
  provider_input TEXT,
  event_type_input TEXT,
  payload_input JSONB
) RETURNS JSONB
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  result_record RECORD;
BEGIN
  INSERT INTO public.webhook_events (
    provider,
    event_type,
    payload,
    processed,
    created_at
  ) VALUES (
    provider_input,
    event_type_input,
    payload_input,
    FALSE,
    NOW()
  )
  RETURNING id INTO result_record;
  
  RETURN jsonb_build_object('id', result_record.id);
END;
$$;

-- 2. Função para atualizar o status do evento de webhook
CREATE OR REPLACE FUNCTION public.update_webhook_event_status(
  event_id_input UUID,
  processed_input BOOLEAN,
  processing_result_input TEXT
) RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  UPDATE public.webhook_events
  SET
    processed = processed_input,
    processed_at = NOW(),
    processing_result = processing_result_input
  WHERE id = event_id_input;
END;
$$;

-- 5. Função para atualizar status da assinatura
CREATE OR REPLACE FUNCTION public.update_subscription_status(
  subscription_id_input TEXT,
  status_input TEXT
) RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  UPDATE public.subscriptions
  SET
    status = status_input,
    updated_at = NOW()
  WHERE external_id = subscription_id_input;
END;
$$;

-- 6. Função para encontrar perfil por customer_id
CREATE OR REPLACE FUNCTION public.find_profile_by_customer_id(
  customer_id_input TEXT
) RETURNS JSONB
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  profile_record RECORD;
BEGIN
  SELECT id INTO profile_record
  FROM public.profiles
  WHERE external_customer_id = customer_id_input;
  
  IF profile_record IS NULL THEN
    RETURN NULL;
  END IF;
  
  RETURN jsonb_build_object('id', profile_record.id);
END;
$$;

-- 7. Função para encontrar perfil por ID

-- 8. Função para atualizar customer_id do perfil
CREATE OR REPLACE FUNCTION public.update_profile_customer_id(
  profile_id_input UUID,
  customer_id_input TEXT
) RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  UPDATE public.profiles
  SET external_customer_id = customer_id_input

$$;

-- 9. Função para atualizar status da assinatura do perfil

-- 10. Função para atualizar próxima data de cobrança da assinatura

SECURITY DEFINER

-- Além disso, precisamos criar as tabelas necessárias que estão sendo referenciadas no código
-- Você deve verificar se essas tabelas já existem antes de criar

-- 1. Tabela de eventos de webhook (webhook_events)
CREATE TABLE IF NOT EXISTS public.webhook_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  provider TEXT NOT NULL,
  event_type TEXT NOT NULL,
  payload JSONB NOT NULL,
  processed BOOLEAN NOT NULL DEFAULT FALSE,
  processed_at TIMESTAMPTZ,
  processing_result TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 2. Tabela de histórico de pagamentos (payment_history)
CREATE TABLE IF NOT EXISTS public.payment_history (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  external_id TEXT NOT NULL,
  customer_id TEXT NOT NULL,
  amount NUMERIC NOT NULL,
  payment_date TIMESTAMPTZ NOT NULL,
  due_date TIMESTAMPTZ,
  payment_method TEXT NOT NULL,
  description TEXT NOT NULL,
  status TEXT NOT NULL,
  subscription_id TEXT,
  provider TEXT NOT NULL DEFAULT 'asaas',
  invoice_url TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 3. Tabela de assinaturas (subscriptions)
CREATE TABLE IF NOT EXISTS public.subscriptions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  external_id TEXT NOT NULL UNIQUE,
  customer_id TEXT NOT NULL,
  plan_value NUMERIC NOT NULL,
  billing_type TEXT NOT NULL,
  status TEXT NOT NULL,
  next_billing_date TIMESTAMPTZ NOT NULL,
  description TEXT NOT NULL,
  provider TEXT NOT NULL DEFAULT 'asaas',
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 4. Alterar a tabela de perfis para adicionar o campo external_customer_id se não existir
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT FROM information_schema.columns 
    WHERE table_schema = 'public' AND table_name = 'profiles' AND column_name = 'external_customer_id'
  ) THEN
    ALTER TABLE public.profiles ADD COLUMN external_customer_id TEXT;
  END IF;
END $$;
