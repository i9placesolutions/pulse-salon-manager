
-- Função para obter bloqueios de horário para um estabelecimento
CREATE OR REPLACE FUNCTION public.get_blocked_times(
  establishment_id_input UUID
) RETURNS SETOF public.blocked_times
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  RETURN QUERY
  SELECT *
  FROM public.blocked_times
  WHERE establishment_id = establishment_id_input
  ORDER BY start_date ASC;
END;
$$;

-- Função para criar um bloqueio de horário
CREATE OR REPLACE FUNCTION public.create_blocked_time(
  establishment_id_input UUID,
  professional_id_input UUID,
  start_date_input DATE,
  end_date_input DATE,
  start_time_input TIME DEFAULT NULL,
  end_time_input TIME DEFAULT NULL,
  reason_input TEXT DEFAULT NULL,
  is_full_day_input BOOLEAN DEFAULT FALSE
) RETURNS UUID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  new_id UUID;
BEGIN
  INSERT INTO public.blocked_times (
    establishment_id,
    professional_id,
    start_date,
    end_date,
    start_time,
    end_time,
    reason,
    is_full_day,
    created_at
  ) VALUES (
    establishment_id_input,
    professional_id_input,
    start_date_input,
    end_date_input,
    start_time_input,
    end_time_input,
    reason_input,
    is_full_day_input,
    NOW()
  )
  RETURNING id INTO new_id;
  
  RETURN new_id;
END;
$$;

-- Função para excluir um bloqueio de horário
CREATE OR REPLACE FUNCTION public.delete_blocked_time(
  block_id_input UUID
) RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  affected_rows INT;
BEGIN
  DELETE FROM public.blocked_times
  WHERE id = block_id_input
  RETURNING 1 INTO affected_rows;
  
  RETURN affected_rows > 0;
END;
$$;
